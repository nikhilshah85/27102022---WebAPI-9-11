	a. Add a nuget package CORS 
	b. in Program.cs 
		builder.Services.AddCors(policy =>
		{
		    policy.AddDefaultPolicy(defaultPolicy =>
		    {
		        defaultPolicy.AllowAnyHeader().AllowAnyMethod().AllowAnyOrigin();
		    });
		});

	app.UseCors();
